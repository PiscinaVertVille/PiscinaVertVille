// Como aplicar:
// 1. Acesse o Firebase Console > seu projeto > Firestore Database > Regras
// 2. Cole o conteúdo abaixo (substituindo o que já existir)
// 3. Clique em "Publicar"
//
// Modelo de segurança (atualizado): tanto a médica quanto o morador agora
// usam Firebase Authentication (email/senha) de verdade. Isso permite regras
// muito mais restritas baseadas em request.auth.uid, sem depender de
// "escrita aberta para todo mundo".
//
// - moradores/{moradorId}: o documento ID é o próprio uid do Firebase Auth.
//   Cada morador só pode criar/editar o PRÓPRIO documento. Leitura pública
//   (necessária para o cálculo de disponibilidade não depender de login).
// - configExame, plantoesImportados, bloqueiosManuais: leitura pública,
//   escrita só da médica autenticada.
// - agendamentos: leitura pública (necessária pro calendário saber o que
//   está ocupado), criação permitida para qualquer usuário autenticado
//   (morador logado) desde que o moradorId do agendamento seja o uid dele,
//   update/delete só da médica (marcar pago, excluir, resolver conflito).
// ==========================================================================

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function ehMedica() {
      // A médica é o único usuário do Firebase Auth que TAMBÉM não tem
      // documento próprio em /moradores. Como isso é difícil de checar
      // diretamente nas regras, usamos uma lista simples de emails autorizados.
      // Atualize esse email se a conta da médica mudar.
      return request.auth != null &&
             request.auth.token.email == "mrremondini@gmail.com";
    }

    function ehMoradorAutenticado() {
      return request.auth != null;
    }

    function ehProprioMorador(moradorId) {
      return request.auth != null && request.auth.uid == moradorId;
    }

    // Moradores: cada um só cria/edita o próprio documento (uid == moradorId).
    // ATENÇÃO: o documento contém o campo "codigoVerificacao" (sensível).
    // Por simplicidade técnica (sem Cloud Functions), a leitura é pública,
    // mas o código de verificação só fica num documento por poucos minutos
    // até a confirmação, e o impacto de exposição é baixo (não é senha —
    // a senha real fica protegida pelo próprio Firebase Auth, nunca exposta
    // em nenhum documento do Firestore). Leitura pública é necessária para
    // o painel da médica listar moradores e para o cálculo de disponibilidade.
    match /moradores/{moradorId} {
      allow read: if true;
      allow create: if ehProprioMorador(moradorId);
      allow update: if ehProprioMorador(moradorId) || ehMedica();
      allow delete: if ehMedica();
    }

    // Configuração de exames e pacotes: leitura pública, escrita só da médica.
    match /configExame/{docId} {
      allow read: if true;
      allow write: if ehMedica();
    }

    // Plantões importados (backup manual do Shiftr): leitura pública, escrita só da médica.
    match /plantoesImportados/{docId} {
      allow read: if true;
      allow write: if ehMedica();
    }

    // Bloqueios manuais (compromissos pessoais): leitura pública, escrita só da médica.
    match /bloqueiosManuais/{docId} {
      allow read: if true;
      allow write: if ehMedica();
    }

    // Agendamentos: leitura pública, criação por qualquer morador autenticado
    // (desde que o moradorId do documento seja o uid de quem está criando),
    // update/delete só da médica (marcar como pago, excluir, resolver conflito).
    match /agendamentos/{docId} {
      allow read: if true;
      allow create: if ehMoradorAutenticado() && request.resource.data.moradorId == request.auth.uid;
      allow update: if ehMedica();
      allow delete: if ehMedica();
    }
  }
}
